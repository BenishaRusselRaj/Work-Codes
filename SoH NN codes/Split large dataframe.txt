import pandas as pd

f=""
data=pd.read_csv(f)

data=data.reset_index(drop=True)

a=len(data)//2
i=a//2
j=a+((len(data)-a)//2)

d1=data[:i,:]
d2=data[i:a,:]
d3=data[a:j,:]
d4=data[j:,:]

filename=f.rsplit('.',1)[0]
d1.to_csv(filename+'_1.csv')
d2.to_csv(filename+'_2.csv')
d3.to_csv(filename+'_3.csv')
d4.to_csv(filename+'_4.csv')
